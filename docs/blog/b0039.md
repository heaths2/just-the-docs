### 디스크 이미지 저장

```bash
sudo -i
ifconfig eth1 192.168.56.200/24 up
sshfs root@192.168.56.111:/home/partimag /home/partimag -p 22 -o noatime
yes
# 대화형 -x 확인 -c
/usr/sbin/ocs-sr -q2 -c -x -j2 -z1p -i 4096 -sfsck -senc -p reboot savedisk Test sda
# 스킵형
/usr/sbin/ocs-sr -q2 -j2 -z1p -i 4096 -sfsck -senc -p reboot savedisk Test sda
# 디스크 검사
/usr/sbin/ocs-sr -q2 -j2 -z1p -i 4096 -fsck-y -senc -p reboot savedisk Test sda
```

### 디스크 이미지 복원

```bash
sshfs root@192.168.56.111:/home/partimag /home/partimag -p 22 -o noatime
yes
# 대화형 -x 확인 -c
/usr/sbin/ocs-sr -g auto -e1 auto -e2 -r -j2 -x -icds -k1 -p reboot restoredisk Test sda
```

### 이미지 저장 스크립트

```bash
tee Clonezilla_savedisk.sh << EOF
#!/usr/bin/env bash
unset LANG
unset LANGUAGE
LC_ALL=en_US.UTF-8
export LC_ALL

echo "+--------------------------------------------+"
echo "| Please enter the image name:               |"
echo "+--------------------------------------------+"

# Prompt user for input
read -p "Image name: [${img:-Ubuntu}] " img
read -p "Are you sure you want to save the image '${img:-Ubuntu}'? (y/N): " confirm

if [[ "$confirm" =~ ^[Yy]$ ]]; then
    echo "Starting the saving process..."
    # Execute the Clonezilla command
    /usr/sbin/ocs-sr -q2 -j2 -z1p -i 4096 -fsck-y -senc -p reboot savedisk "$img" sda
else
    echo "Operation cancelled."
    exit 0
fi
EOF
```

### 이미지 복구 스크립트

```bash
tee Clonezilla_restoredisk.sh << EOF
#!/usr/bin/env bash
unset LANG
unset LANGUAGE
LC_ALL=en_US.UTF-8
export LC_ALL

echo "+--------------------------------------------+
| Please select one of the following options:|
+--------------------------------------------+
| 1) u24-XEN                                 |
| 2) u22-XEN                                 |
| 3) u24-BareMetal                           |
| 4) u22-BareMetal                           |
+--------------------------------------------+"

# Prompt user for input
read -p "Choose (1/2/3/4): " choice

# Set image variable based on selection
case $choice in
    1)
        echo "You have selected u24-XEN."
        img="u24-XEN"
        ;;
    2)
        echo "You have selected u22-XEN."
        img="u22-XEN"
        ;;
    3)
        echo "You have selected u24-BareMetal."
        img="u24-BareMetal"
        ;;
    4)
        echo "You have selected u22-BareMetal."
        img="u22-BareMetal"
        ;;
    *)
        echo "Invalid input. Exiting the program."
        exit 1
        ;;
esac

# Confirmation message
echo "Selected image: $img"
read -p "Are you sure you want to restore the '$img' image? (y/N): " confirm

if [[ "$confirm" =~ ^[Yy]$ ]]; then
    echo "Starting the restoration process..."
    # Execute Clonezilla command
    /usr/sbin/ocs-sr -g auto -e1 auto -e2 -r -j2 -icds -k1 -p reboot restoredisk "$img" sda
else
    echo "Operation cancelled."
    exit 0
fi
EOF
```
