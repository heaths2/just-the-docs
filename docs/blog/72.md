```bash
#!/bin/bash
set -u
export LC_ALL=C
 
read -rp "서버 목록을 생성하시겠습니까? (y/n): " choice
case "$choice" in
  y|Y)
    read -p "Enter hostname: " name
    read -p "Start number: " s_int
    read -p "End number: " e_int
 
    for i in $(seq -f "%02g" $s_int $e_int); do
        echo "${name}${i}"
    done > serverlist
    echo "Server list generated."
    ;;
  n|N)
    echo "Server list generation cancelled."
    ;;
  *) echo "Invalid input. Server list generation cancelled." ;;
esac
```

```bash
#!/bin/bash
set -u
export LC_ALL=C
 
function print_banner() {
  # 최소 테이블 헤더
  local MAX=16
  local dash=$(printf -- '-%.0s' $(seq 1 $((MAX - 2))))
 
  while true; do
    read -rp "데이터를 입력하시겠습니까? (y/n): " choice </dev/tty
 
    case "$choice" in
      [yY])
        # 사용자 입력 받기
        read -rp "Enter comma-separated data: " input_data </dev/tty
 
        if [[ -n "$input_data" ]]; then
          # 쉼표를 기준으로 데이터를 분리하여 배열에 저장
          IFS=',' read -ra table_data <<< "$input_data"
 
          for line in "${table_data[@]}"; do
              if [[ ${#line} -gt $MAX ]]; then
                MAX=${#line}
              fi
          done
 
          # 테이블 헤더
          printf "+${dash}+\n"
          printf "| %-${MAX}s |\n" "$(cat /etc/hostname)"
          printf "+${dash}+\n"
          # 테이블 데이터 출력
          for data in "${table_data[@]}"; do
            printf "| %-${MAX}s |\n" "$data"
          done
          # 테이블 푸터
          printf "+${dash}+\n"
        fi
        break
        ;;
      [nN])
        # 테이블 헤더
        printf "+${dash}+\n"
        printf "| %-${MAX}s |\n" "$(cat /etc/hostname)"
        printf "+${dash}+\n"
        break
        ;;
      *)
        echo "Invalid input. Please enter again."
        ;;
    esac
  done
 
}
 
print_banner
```
